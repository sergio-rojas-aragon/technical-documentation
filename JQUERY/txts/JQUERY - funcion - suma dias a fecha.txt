suma cantidad indicada de dias a una fecha en un input ***			function calculafecha(diasumar){		if (diasumar == 0){			if (jQuery('#condpago').val() == '001'){				jQuery('#fvenci').val(jQuery('#fecha').val());			}else{				jQuery('#fvenci').val(jQuery('#fvenci_original').val());			}		}else{			fechaux = jQuery('#fecha').val();			cadena = fechaux.split('/');			temp1 = cadena[1] +"/"+ cadena[0] +"/"+ cadena[2];			var cfd = Date.parse(temp1);			hoy = new Date(cfd);			for (i=0; i<diasumar; i++){				hoy.setTime(hoy.getTime()+24*60*60*1000); // añadimos 1 día				mes = hoy.getMonth()+1;				dia = hoy.getDate();			}			if (mes<10) mes = '0'+mes;			if (dia<10) dia = '0'+dia;				fecha = dia + '/' + mes + '/' + hoy.getFullYear();			jQuery('#fvenci').val(fecha);		}	}
